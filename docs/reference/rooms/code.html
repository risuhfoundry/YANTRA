import numpy as np class NeuralEngine: def __init__(self, layers): self.layers = layers self.history = [] def backward_pass(self, loss_gradient): # TODO: Implement Gradient Clipping max_norm = 1.0 # Calculate global norm total_norm = np.sqrt(sum(np.sum(np.square(g)) for g in self.gradients)) if total_norm > max_norm: clip_coeff = max_norm / (total_norm + 1e-6) for g in self.gradients: g *= clip_coeff return self.apply_updates()
